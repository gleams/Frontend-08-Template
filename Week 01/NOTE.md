 let pattern = [
         [2,0,0],
         [0,1,0],
         [0,0,0]
     ];

   console.log(pattern[0][0]) //输出2
   //取1
   console.log(pattern[1][1])//输出1

   //          0,1,0

   //          j

   //          0,2,0
   //          0,0,2
   
   
   // 对角线第一种情况为 i,j相等。
   //          j
   //        i 0,0,2
   //          0,2,0
   //          2,0,0
   
   // 对角线第二种情况为 i,j相加等于2。也就是i,j相加等于内层的长度减1
   
   //          j
   //        i 0,0,0,2
   //          0,0,2,0
   //          0,2,0,0
   //          2,0,0,0
